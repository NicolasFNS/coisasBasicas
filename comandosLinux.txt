11- chmod /* modifica as permissões de acesso de arquivos e diretórios */ - /* chmod u=rwx, g=rw, o=r arquivo.sh */ /* chmod 755 arquivo.sh */
  /* rwx quer dizer read, write e execute, são as permissôes e as letras são os 3 tipos de usuário: User, Group e Other */
  
  /* os 3 números representam os usuários que mencionei na mesma sequência do anterior U, G e O 
  e o valor é o nível de permissão: 0 = sem permissão, 1 = somente execução, 2 = somente grava e 4 = somente ler
  para ter mais de uma permissão em um grupo, some os valores das permissões, por exemplo 3 = ler e gravar ou 5 = executar e ler */
